54  {

56  // A = result matrix
57  // n = number of rows = number of columns in A (n x n)
58  int pivrow; // keeps track of current pivot row
59  int k, i, j; // k: overall index along diagonal; i: row index; j: col index
60  int pivrows[n]; // keeps track of rows swaps to undo at end
61  double tmp; // used for finding max value and making column swaps
62 
63  for (i = 0 ; i < n; i++)
64  for (j = 0 ; j < n; j++)
65  A[i * n + j] = C[i * n + j];
66 
67  for (k = 0; k < n; k++)
68  {
69  // find pivot row, the row with biggest entry in current column
70  tmp = 0;
71  for (i = k; i < n; i++)
72  {
73  if (abs(A[i * n + k]) >= tmp) // 'Avoid using other functions inside abs()?'
74  {
75  tmp = abs(A[i * n + k]);
76  pivrow = i;
77  }
78  }
79 
80  // check for singular matrix
81  if (A[pivrow * n + k] == 0.0f)
82  {
83  return -1;
84  }
85 
86  // Execute pivot (row swap) if needed
87  if (pivrow != k)
88  {
89  // swap row k with pivrow
90  for (j = 0; j < n; j++)
91  {
92  tmp = A[k * n + j];
93  A[k * n + j] = A[pivrow * n + j];
94  A[pivrow * n + j] = tmp;
95  }
96  }
97  pivrows[k] = pivrow; // record row swap (even if no swap happened)
98 
99  tmp = 1.0f / A[k * n + k]; // invert pivot element
100  A[k * n + k] = 1.0f; // This element of input matrix becomes result matrix
101 
102  // Perform row reduction (divide every element by pivot)
103  for (j = 0; j < n; j++)
104  {
105  A[k * n + j] = A[k * n + j] * tmp;
106  }
107 
108  // Now eliminate all other entries in this column
109  for (i = 0; i < n; i++)
110  {
111  if (i != k)
112  {
113  tmp = A[i * n + k];
114  A[i * n + k] = 0.0f; // The other place where in matrix becomes result mat
115  for (j = 0; j < n; j++)
116  {
117  A[i * n + j] = A[i * n + j] - A[k * n + j] * tmp;
118  }
119  }
120  }
121  }
122 
123  // Done, now need to undo pivot row swaps by doing column swaps in reverse order
124  for (k = n - 1; k >= 0; k--)

129  {

132  A[i * n + pivrows[k]] = tmp;
133  }
134  }
135  }
